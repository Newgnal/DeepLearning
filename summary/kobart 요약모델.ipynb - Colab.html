model_name = "digit82/kobart-summarization"

model = BartForConditionalGeneration.from_pretrained(model_name)

for p in model.model.encoder.parameters():
    p.requires_grad = False

# 4. 데이터 불러오기 (train.csv / val.csv는 미리 생성돼 있어야 함)
train_df = pd.read_csv("train.csv")
val_df = pd.read_csv("val.csv")

train_dataset_raw = HFDataset.from_pandas(train_df)
val_dataset_raw = HFDataset.from_pandas(val_df)

def tokenize(batch):
    inputs = tokenizer(batch["text"], padding="max_length", truncation=True, max_length=384)
    targets = tokenizer(batch["summary"], padding="max_length", truncation=True, max_length=64)
    return {
        "input_ids": inputs["input_ids"],
        "attention_mask": inputs["attention_mask"],
        "labels": targets["input_ids"]
    }

# 6. 토큰화 적용
tokenized_train = train_dataset_raw.map(tokenize, batched=True, remove_columns=["text", "summary"])

# 7. PyTorch Dataset 및 DataLoader 정의
class SummaryDataset(Dataset):
    def __init__(self, dataset):
        self.dataset = dataset

    def __len__(self):
        return len(self.dataset)

    def __getitem__(self, idx):
        return {k: torch.tensor(v) for k, v in self.dataset[idx].items()}

train_dataset = SummaryDataset(tokenized_train)
val_dataset = SummaryDataset(tokenized_val)

class KoBARTFineTuner(pl.LightningModule):
    def __init__(self, model):
        super().__init__()
        self.save_hyperparameters(ignore=["model"])  # 체크포인트 오류 방지
        self.model = model

    def forward(self, input_ids, attention_mask, labels=None):
        return self.model(input_ids=input_ids, attention_mask=attention_mask, labels=labels)

    def training_step(self, batch, batch_idx):
        outputs = self(**batch)
        self.log("train_loss", outputs.loss, prog_bar=True)
        return outputs.loss

    def validation_step(self, batch, batch_idx):
        outputs = self(**batch)
        self.log("val_loss", outputs.loss, prog_bar=True)

    def configure_optimizers(self):
        return AdamW(self.model.parameters(), lr=5e-5)

    limit_val_batches=1.0

import pandas as pd
import torch
from transformers import PreTrainedTokenizerFast, BartForConditionalGeneration
from pytorch_lightning import LightningModule
from tqdm import tqdm
import glob
import os
import re
import warnings
warnings.filterwarnings("ignore")

# 1. KoBART 파인튜너 클래스 정의
class KoBARTFineTuner(LightningModule):
    def __init__(self, model):
        super().__init__()
        self.model = model

    def forward(self, input_ids, attention_mask, labels=None):
        return self.model(input_ids=input_ids, attention_mask=attention_mask, labels=labels)

# 2. 체크포인트 경로 자동 탐색
ckpt_list = sorted(
    glob.glob("/content/lightning_logs/version_*/checkpoints/*.ckpt"),
    key=os.path.getmtime
)
if not ckpt_list:
    raise FileNotFoundError("체크포인트 파일을 찾을 수 없습니다. 학습이 완료되었는지 확인하세요.")

ckpt_path = ckpt_list[-1]
print(f"최신 체크포인트 선택됨: {ckpt_path}")

# 3. 토크나이저 및 모델 로딩
tokenizer = PreTrainedTokenizerFast.from_pretrained("digit82/kobart-summarization")
base_model = BartForConditionalGeneration.from_pretrained("digit82/kobart-summarization")
model_module = KoBARTFineTuner.load_from_checkpoint(ckpt_path, model=base_model)
model = model_module.model.to("cuda" if torch.cuda.is_available() else "cpu")

# 4. 뉴스 데이터 불러오기 → 상위 300개만 사용
df = pd.read_csv("news_bal_30000_cleaned_for_electra.csv")
df = df.dropna(subset=["clean_text"]).iloc[:300].copy()  # ← 여기만 자름
texts = df["clean_text"].astype(str).tolist()

# 5. 요약 함수
def summarize(text, max_len=150):
    input_ids = tokenizer.encode(
        text, return_tensors="pt", truncation=True, max_length=384
    ).to(model.device)

    summary_ids = model.generate(
        input_ids,
        max_length=max_len,
        num_beams=4,
        early_stopping=True,
        no_repeat_ngram_size=2
    )
    summary = tokenizer.decode(summary_ids[0], skip_special_tokens=True)

    sentences = re.split(r'(?<=[.?!])\s+', summary.strip())
    trimmed = " ".join(sentences[:3]).strip()
    return trimmed if trimmed else "요약 실패"

# 6. 전체 요약 수행
summaries = []
errors = []

for i, text in enumerate(tqdm(texts, desc="📄 뉴스 요약 중", leave=True)):
    try:
        summary = summarize(text)
    except Exception as e:
        summary = "요약 실패"
        errors.append((i, str(e)))
    summaries.append(summary)

# 7. 저장
df["generated_summary"] = summaries
df.to_csv("news_sample_300_with_summary.csv", index=False)
print("뉴스 요약 완료 → news_sample_300_with_summary.csv")

# 8. 실패 건 출력 (선택)
if errors:
    print(f"요약 실패 건수: {len(errors)}")
    for idx, msg in errors[:5]:  # 최대 5개만 출력
        print(f"  - [{idx}] 에러: {msg}")
else:
    print("모든 뉴스 요약 성공!")
